    <script>
        document.addEventListener('DOMContentLoaded', getQuoteOfTheDay);

        async function getQuoteOfTheDay() {
            const quoteContentDiv = document.getElementById('quote-content');
            quoteContentDiv.innerHTML = '<p class="loading">명언을 불러오는 중...</p>';

            try {
                const response = await fetch('https://quotes.rest/qod?category=inspire');

                if (!response.ok) {
                    throw new Error(`HTTP 오류 발생! 상태 코드: ${response.status}`);
                }

                const data = await response.json();

                if (data && data.contents && data.contents.quotes && data.contents.quotes.length > 0) {
                    const quote = data.contents.quotes[0];
                    quoteContentDiv.innerHTML = `
                        <p id="quote-text">"${quote.quote}"</p>
                        <p id="quote-author">- ${quote.author} -</p>
                    `;
                } else {
                    throw new Error('API 응답에서 명언 데이터를 찾을 수 없습니다.');
                }

            } catch (error) {
                console.error('명언을 가져오는 중 오류 발생:', error);
                quoteContentDiv.innerHTML = `<p class="error">명언을 가져오는 데 실패했습니다.<br>다시 시도해 주세요. (${error.message})</p>`;
            }
        }
    </script>
